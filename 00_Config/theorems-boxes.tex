% config/theorems-boxes.tex — 定理环境与 tcolorbox 盒子
% ======================================================

% ===== tcolorbox 基础样式 =====
\tcbset{
  basestyle/.style={
    enhanced,
    breakable,
    sharp corners,
    left=8pt, right=8pt, top=8pt, bottom=8pt,
    boxrule=0.8pt,
    before skip=10pt plus 2pt,
    after skip=10pt plus 2pt,
    fonttitle=\bfseries,
  },
}

% =================================================================
%  编号定理类环境（使用 tcolorbox theorem 库）
%  用法: \begin{xxxbox}{标题}{label-key} 内容 \end{xxxbox}
%  引用: \cref{prefix:label-key}
% =================================================================

% ----- 定义 (def:) -----
\newtcbtheorem[number within=chapter]{definitionbox}{定义}{%
  basestyle,
  colframe=BoxDefFrame,
  colback=BoxDefBack,
  colbacktitle=BoxDefFrame,
  coltitle=white,
  attach boxed title to top left={yshift=-2mm,xshift=4mm},
  boxed title style={sharp corners, boxrule=0pt},
}{def}
